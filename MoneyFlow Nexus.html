คู่มือการติดตั้งด้วยตนเอง

คู่มือนี้จะพาคุณติดตั้งระบบทั้งหมดภายใน 10-15 นาที โดยใช้ทรัพยากรฟรีจาก Google และ GitHub

1

📁 สร้าง Google Sheets และ Apps Script

1. ไปที่ sheets.new สร้างชีตใหม่
2. เปลี่ยนชื่อ Sheet เป็น Transactions, Wallets, ExpenseCats, IncomeCats, Settings
3. ไปที่ ส่วนขยาย (Extensions) > Apps Script

function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet(); // จัดการ action ต่างๆ return ContentService.createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } }

4. กด บันทึก > ทำให้ใช้งานได้ (Deploy) > New Deployment > Web App
5. ตั้งค่า Execute as: Me, Who has access: Anyone > กด Deploy และ คัดลอก URL

2

🔗 เชื่อมต่อ URL ในโค้ด HTML

เปิดไฟล์ index.html และค้นหาบรรทัด:

const NEXO_API_URL = 'YOUR_SCRIPT_URL_HERE';

นำ URL ที่คัดลอกมาจากขั้นตอนที่ 1 มาใส่แทนที่ YOUR_SCRIPT_URL_HERE
หากแก้ไข Apps Script ต้อง Deploy > New Version และอัปเดต URL ด้วย

3

🌍 อัปโหลดขึ้น GitHub Pages

1. สร้าง Repository ใหม่บน GitHub
2. อัปโหลดไฟล์ index.html ขึ้นไปที่ Repository
3. ไปที่ Settings > Pages
4. เลือก Branch main แล้วกด Save
5. รอ 1-2 นาที จะได้ลิงก์ https://username.github.io/moneyflow-app

4

📊 ตั้งค่าข้อมูลเริ่มต้น

เปิด Apps Script และรันฟังก์ชัน initSheets() เพื่อสร้าง Header และข้อมูลตัวอย่าง

function initSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // สร้าง Transactions sheet let sheet = ss.getSheetByName('Transactions') || ss.insertSheet('Transactions'); sheet.getRange('A1:H1').setValues([['id','type','amount','catId','walletId','date','time','note']]); // สร้าง Wallets sheet sheet = ss.getSheetByName('Wallets') || ss.insertSheet('Wallets'); sheet.getRange('A1:D1').setValues([['id','name','icon','color']]); sheet.getRange('A2:D4').setValues([ ['w1','กระเป๋าหลัก','fa-wallet','#6c63ff'], ['w2','บัตรเครดิต','fa-credit-card','#8b5cf6'], ['w3','ออมทรัพย์','fa-piggy-bank','#00d68f'] ]); console.log('✅ Initial setup complete!'); }
